worker_processes  1;



events {
    worker_connections  1024;
}


http {
    include       mime.types;
    default_type  application/octet-stream;


    sendfile        on;

    keepalive_timeout  65;


    upstream backend {
        server localhost:8080;
    }

    server {
        listen       80;
        server_name  localhost;


        location =/signin/{
            root   html;
            index form.html;
        }

        location /{
            proxy_pass http://backend/;
        }

        error_page   500 502 503 504  /50x.html;
    }
}
